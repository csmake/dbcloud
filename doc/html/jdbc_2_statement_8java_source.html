83  private ArrayList<Document> getColumns(ResultSet rs) throws SQLException {
84  ResultSetMetaData columns = rs.getMetaData();
85  int columnsCount = columns.getColumnCount();
86  ArrayList<Document> header = new ArrayList();
87  for (int i = 1; i <= columnsCount; i++) {
88  Document doc = new Document();
89  String name = columns.getColumnName(i);
90  doc.append("name", name);
91  String label = columns.getColumnLabel(i);
92  if (label != null && !label.isEmpty()) {
93  doc.append("label", label);
94  }
95  String className = columns.getColumnClassName(i);
96  if (className != null && !className.isEmpty()) {
97  doc.append("className", className);
98  }
99  String typeName = columns.getColumnTypeName(i);
100  if (typeName != null && !typeName.isEmpty()) {
101  doc.append("typeName", typeName);
102  }
103  String schemaName = columns.getSchemaName(i);
104  if (schemaName != null && !schemaName.isEmpty()) {
105  doc.append("schemaName", schemaName);
106  }
107  String tableName = columns.getTableName(i);
108  if (tableName != null && !tableName.isEmpty()) {
109  doc.append("tableName", tableName);
110  }
111  header.add(doc);
112  }
113  return header;

116  private org.dbcloud.jdbc.ResultSet convResultSet(ResultSet rs, org.dbcloud.jdbc.ResultSet result, Document options) throws SQLException {
117  if (rs != null) {
118  if (options.getBoolean("document", false)) {
119  ArrayList<Object> list = new ArrayList();
120  ResultSetMetaData columns = rs.getMetaData();
121  result.columns=getColumns(rs);
122  int columnsCount = columns.getColumnCount();
123  while (rs.next()) {
124  Document doc = new Document();
125  // read the result set
126  for (int i = 1; i <= columnsCount; i++) {
127  String name = columns.getColumnName(i);
128  doc.append(name, rs.getObject(i));
129  }
130  list.add(doc);
131  }
132  result.rows= list;
133  } else {
134  ArrayList columns = getColumns(rs);
135  int columnsCount = columns.size();
136  result.columns= columns;
137  ArrayList<Object> rows = new ArrayList();
138  while (rs.next()) {
139 
140  // read the result set
141  ArrayList<Object> cells = new ArrayList();
142  for (int i = 1; i <= columnsCount; i++) {
143  cells.add(rs.getObject(i));
144  }
145  rows.add(cells);
146  }
147  result.rows=rows;
148  }
149  rs.close();
150  rs = null;
151  }
152  return result;
153  }
