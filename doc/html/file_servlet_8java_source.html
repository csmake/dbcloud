64  @Override
65  protected void doGet(HttpServletRequest request, HttpServletResponse response)
66  throws ServletException, IOException {
67  String id = request.getParameter("id");
68  String name = request.getParameter("name");
69  String sql = null;
70  if (id != null) {
71  sql = String.format(this.getInitParameter("selectById"), id);
72  } else if (name != null) {
73  sql = String.format(this.getInitParameter("selectByName"), name);
74  }
75 
76  if (sql != null) {
77  org.dbcloud.jdbc.ResultSet rst = stmt.executeQuery(sql, new Document("document", true));
78  ArrayList rows = rst.rows;
79  if (rows.size() > 0) {
80  Document file = (Document) rows.get(0);
81  String type = file.containsKey("contentType") ? file.get("contentType").toString() : "application/octet-stream";
82  response.setContentType(type);
83  long fileLength = file.getLong("length"); // 记录文件大小
84  long pastLength = 0; // 记录已下载文件大小
85  long contentLength = 0; // 客户端请求的字节总量
86  String contentRange = "";
87  if (request.getHeader("Range") != null) { // 客户端请求的下载的文件块的开始字节
88  response.setStatus(javax.servlet.http.HttpServletResponse.SC_PARTIAL_CONTENT);
89  long toLength = 0; // 记录客户端需要下载的字节段的最后一个字节偏移量（比如bytes=27000-39000，则这个值是为39000）
90  String rangeBytes = request.getHeader("Range").replaceAll("bytes=", "");// 记录客户端传来的形如“bytes=27000-”或者“bytes=27000-39000”的内容
91  String[] tmps = rangeBytes.split("-");
92  pastLength = Long.parseLong(tmps[0].trim());
93  if (tmps.length == 2) {
94  String str = tmps[1].trim();
95  toLength = str.length() > 0 ? Long.parseLong(str) : fileLength;
96  contentLength = toLength - pastLength;
97  toLength = pastLength + contentLength;
98  } else {
99  contentLength = fileLength - pastLength;
100  toLength = pastLength + contentLength;
101  }
102  contentRange = new StringBuffer("bytes ").append(Long.toString(pastLength)).append("-").append(Long.toString(toLength - 1)).append("/").append(Long.toString(fileLength)).toString();
103  response.setHeader("Accept-Ranges", "bytes");// 如果是第一次下,还没有断点续传,状态是默认的 200,无需显式设置;响应的格式是:HTTP/1.1 200 OK
104  response.setHeader("Content-Range", contentRange);
105  response.addHeader("Content-Length", String.valueOf(contentLength));
106  response.setHeader("Content-Disposition", "attachment;filename="+file.getString("name"));
107  } else { // 从开始进行下载
108  contentLength = fileLength; // 客户端要求全文下载
109  }
110 
111  try (OutputStream out = response.getOutputStream()) {
112  try (java.io.InputStream is = new java.io.FileInputStream(String.format("%s/%s", root, file.getString("path")))) {
113  if (pastLength != 0) {
114  is.skip(pastLength);
115  }
116  byte[] buffer = new byte[32 * 1024];
117  while (contentLength > 0) {
118  int len = contentLength > buffer.length ? buffer.length : (int) contentLength;
119  len = is.read(buffer, 0, len);
120  out.write(buffer, 0, len);
121  contentLength -= len;
122  }
123  } catch (Exception ex) {
124  System.err.println(ex.getMessage());
125  }
126  }
127  }
128  } else {
129  response.setContentType("text/html;charset=UTF-8");
130  }

173  type = type.replace(".", "image/");
174  } else if (type.endsWith(".mp4") || type.endsWith(".ogg") || type.endsWith(".avi") || type.endsWith(".rmvb") || type.endsWith(".wmv") || type.endsWith(".rm")) {
175  type = type.replace(".", "video/");
